判断字符串中某一特定字符出现的次数

(1)利用分割获取长度原理

/*利用分割获取长度原理*/
console.info("------------------------------------------------");
function getIncludesLength(str,s) {
    return str.split(s).length - 1;
}
var str1 = "hello world! hello javascript!";
var len1 = getIncludesLength(str1,"l");
console.info(len1); //5
console.info("------------------------------------------------");
